  Implementation:
  ```kotlin
  @Test
  fun `Python signed message validates in Kotlin`() {
      // Create message components
      val content = "Python signed message"
      val title = ""
      val timestamp = System.currentTimeMillis() / 1000.0

      // Get Python to compute hash and sign
      val hashResult = python(
          "lxmf_hash",
          "destination_hash" to destDestination.hash.toHex(),
          "source_hash" to sourceDestination.hash.toHex(),
          "timestamp" to timestamp,
          "title" to title,
          "content" to content
      )
      val messageHash = hashResult.getBytes("message_hash")

      // Reconstruct signedPart (same as Kotlin does)
      // Need to get packed payload from Python for exact match
      val packResult = python(
          "lxmf_pack",
          "destination_hash" to destDestination.hash.toHex(),
          "source_hash" to sourceDestination.hash.toHex(),
          "timestamp" to timestamp,
          "title" to title,
          "content" to content,
          "private_key" to testSourceIdentity.sigPrv.toHex()  // 32-byte Ed25519 seed
      )
      val signature = packResult.getBytes("signature")
      val packedPayload = packResult.getBytes("packed_payload")

      // Reconstruct signed_part
      val hashedPart = destDestination.hash + sourceDestination.hash + packedPayload
      val signedPart = hashedPart + messageHash

      // Verify in Kotlin
      testSourceIdentity.validate(signature, signedPart) shouldBe true
  }
  ```

  Note: If `lxmf_pack` doesn't return `packed_payload` separately, may need to:
  1. Add new bridge command `lxmf_sign` that returns signature for given components
  2. Or use `identity_sign` directly with reconstructed signedPart

If bridge doesn't have exact command needed, document what's missing and use alternative approach (e.g., use identity_sign with manually constructed signed_part).

fun `missing source identity causes SOURCE_UNKNOWN`() {
    // Create and pack message
    val message = createTestMessage(content = "Test")
    val packed = message.pack()

    // Clear identity cache (or use fresh identity not in cache)
    val freshSourceIdentity = Identity.create()
    val freshSourceDest = Destination.create(
        identity = freshSourceIdentity,
        direction = DestinationDirection.IN,
        type = DestinationType.SINGLE,
        appName = LXMFConstants.APP_NAME,
        "delivery"
    )

    // Create message with fresh source (not remembered)
    val unrememberedMessage = LXMessage.create(
        destination = destDestination,
        source = freshSourceDest,
        content = "Unremembered source"
    )
    val unrememberedPacked = unrememberedMessage.pack()

    // Attempt to unpack - should have SOURCE_UNKNOWN

    unpacked?.signatureValidated shouldBe false
    unpacked?.unverifiedReason shouldBe UnverifiedReason.SOURCE_UNKNOWN
}
